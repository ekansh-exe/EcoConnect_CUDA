# **🧩 4. Image Fetch + Preprocess (utils.py)**

import requests

from PIL import Image

from io import BytesIO

import cv2

import numpy as np

def fetch_image_from_url(url):

response = requests.get(url)

image = Image.open(BytesIO(response.content)).convert("RGB")

return image

def check_blur(image):

img = np.array(image)

gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

variance = cv2.Laplacian(gray, cv2.CV_64F).var()

return variance > 100 # threshold

## **🟣 utils.py (Image Handling + Blur Detection)**

import requests

from PIL import Image

from io import BytesIO

import cv2

import numpy as np

def fetch_image_from_url(url):

response = requests.get(url, timeout=5)

image = Image.open(BytesIO(response.content)).convert("RGB")

return image

def check_blur(image):

img = np.array(image)

gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

variance = cv2.Laplacian(gray, cv2.CV_64F).var()

return variance > 100 # threshold

import requests

from PIL import Image

from io import BytesIO

import cv2

import numpy as np

def fetch_image_from_url(url):

response = requests.get(url)

image = Image.open(BytesIO(response.content)).convert("RGB")

return image

def check_blur(image):

img = np.array(image)

gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

variance = cv2.Laplacian(gray, cv2.CV_64F).var()

return variance > 100 # threshold

import requests

from PIL import Image

from io import BytesIO

import cv2

import numpy as np

def fetch_image_from_url(url):

response = requests.get(url, timeout=5)

image = Image.open(BytesIO(response.content)).convert("RGB")

return image

def check_blur(image):

img = np.array(image)

gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

variance = cv2.Laplacian(gray, cv2.CV_64F).var()

return variance > 100 # threshold
